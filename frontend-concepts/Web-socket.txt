// socket.js - Initial connection 
import { io } from "socket.io-client";

export const socket = io("https://your-api.com", {
  autoConnect: false,
  transports: ["websocket"],
});


// useSocket.js (custom hook)
import { useEffect } from "react";
import { socket } from "./socket";

export const useSocketConnection = (token, userId) => {
  useEffect(() => {
    if (!token) return;

    socket.auth = { token, userId };
    socket.connect();

    return () => {
      socket.disconnect();
    };
  }, [token, userId]);
};

// Keep connection alive
useEffect(() => {
  const interval = setInterval(() => {
    if (socket.connected) {
      socket.emit("ping_check");
    }
  }, 30000); // 30 sec

  return () => clearInterval(interval);
}, []);


// requestAnimationFrame throttle
// Max 1 update per frame ~16ms
// We used requestAnimationFrame batching to align socket 
//   updates with the browser paint cycle.
import { useEffect, useRef } from "react";
import { socket } from "./socket";

export const useTicketUpdates = (onUpdate) => {
  const frameRef = useRef(null);
  const pendingRef = useRef(false);

  useEffect(() => {
    const handler = (payload) => {
      // mark pending
      pendingRef.current = payload;

      // already scheduled → skip
      if (frameRef.current) return;

      frameRef.current = requestAnimationFrame(() => {
        onUpdate(pendingRef.current);

        frameRef.current = null;
        pendingRef.current = false;
      });
    };

    socket.on("ticket_update", handler);

    return () => {
      socket.off("ticket_update", handler);
      if (frameRef.current) cancelAnimationFrame(frameRef.current);
    };
  }, [onUpdate]);
};


// Room based subscription 
// join relevant room
useEffect(() => {
  if (!userId) return;

  socket.emit("join_room", {
    room: `user_${userId}`,
  });

  return () => {
    socket.emit("leave_room", {
      room: `user_${userId}`,
    });
  };
}, [userId]);


// NodeJS 
io.on("connection", (socket) => {
  socket.on("join_room", ({ room }) => {
    socket.join(room);
  });

  socket.on("leave_room", ({ room }) => {
    socket.leave(room);
  });
});
